 11    def __init__(self, root, callback = lambda : print("Event"), timesleep = 500) :
 12        """Initialise le gestionnaire d'évènement
 13
 14        :param root: Widget parent de notre boucle
 15        :type root: tk.Widget
 16        :param callback: Expression lambda ou fonction devant être exécuté à chaque boucle., defaults to print("Event")
 17        :type callback: lambda|function, optional
 18        :param timesleep: Temps en milliseconde entre chaque exécution., defaults to 500
 19        :type timesleep: int, optional
 20        """
 21        self.root = root
 22        self.function = callback
 23        self.timesleep = timesleep

 26        """Lance l'exécution de la première boucle après un premier interval de la méthode
 27        """
 28        self.root.after(self.timesleep, LoopEvent.prepareCallback(LoopEvent.__loop, self, self.function))

 31        """Lance immédiatement une première exécution de la méthode
 32        """
 33        LoopEvent.__loop(self, self.function)

 37        """Permet de lancer la méthode et génère une nouvelle boucle

 49        """Fonction de préparation d'un callback pour une boucle d'évènement
